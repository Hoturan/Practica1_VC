im = imread('I:\vc\sample images\rabbit.jpg');
sobv = fspecial('sobel')
sobv = sobv/4;
Gy=imfilter(double(im), sobv, 'conv', 'replicate');
Gx=imfilter(double(im), sobv', 'conv', 'replicate');
mod = sqrt(Gx.*Gx + Gy.*Gy);
figure, imshow(mod, [])
bw = mod > 10;
figure, imshow(bw), title('thr=10')
figure, surf(mod)
res = edge(im, 'Canny',[0.1 0.2], 2);
